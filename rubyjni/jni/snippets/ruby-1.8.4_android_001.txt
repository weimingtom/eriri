#ifdef ANDROID
#include <android/log.h>
#endif

static VALUE
rb_f_printf(argc, argv)
    int argc;
    VALUE argv[];
{
    VALUE out;

    if (argc == 0) return Qnil;
    if (TYPE(argv[0]) == T_STRING) {
	out = rb_stdout;
    }
    else {
	out = argv[0];
	argv++;
	argc--;
    }
#ifdef ANDROID    
    //fprintf(stdout, "rb_f_printf : %s\n", RSTRING(rb_f_sprintf(argc, argv))->ptr);
    __android_log_print(ANDROID_LOG_INFO, "ruby_io.c", "%s", RSTRING(rb_f_sprintf(argc, argv))->ptr);
#endif
    rb_io_write(out, rb_f_sprintf(argc, argv));

    return Qnil;
}
